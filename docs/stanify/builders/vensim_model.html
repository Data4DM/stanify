 48class StateVectorIndexMap:
 49    """
 50    This class is used as an element for indicating the map between the indices of a 1d-array to multiple named
 51    nd-arrays. Its usage will be normally be through a list `list[StateVectorIndexMap]`, declared as
 52    `VensimModelContext.state_vector_index_map`. Then, each index of the
 53    `StateVectorIndexMap` in the list, will correspond to the actual variable name and its subscript index held by it.
 54    For example, if we had `[None, StateVectorIndexMap(name="A", indices=(0, 0)]`, this means that the 1st
 55    (note that we don't use the 0th element since stan starts arrays with index 1) element corresponds to the variable
 56    named "A". But since A is subscripted with two subscripts, the indices attribute tells us that it's the 0th value
 57    of the zero-th and first subscript.
 58
 59    Attributes
 60    ----------
 61    name : str
 62        Variable name
 63    indices : tuple[int]
 64        If the variable has subscripts, this corresponds to the index of subscript values the index of the 1d-array
 65        is mapped to. The specific order within the tuple follows `VensimVariableContext.subscripts`
 66    """
 67    name: str
 68    indices: tuple[int]

 71class VensimModelContext:
 72    """
 73    This class holds information about the Vensim model.
 74
 75    Attributes
 76    ----------
 77    _abstract_model : pysd.translators.structures.abstract_model.AbstractModel
 78        The `pysd` abstract model
 79    _first_section : pysd.translators.structures.abstract_model.AbstractSection
 80        The first `AbstractSection` defined within the model. This is purely for convenience and equivalent to
 81        `self._abstract_model.sections[0]`
 82    variables : dict[str, VensimVariableContext]
 83        The dict containing non-stock variables. The variable names are **converted to
 84        identifiers**
 85    integ_outcome_variables : dict[str, VensimVariableContext]
 86        Dict of stock variables. Note that if a variable is a stock variable, it's included in both
 87        `integ_outcome_vector_names` and `variables`
 88    subscripts : dict[str, tuple[Any]]
 89        Values of the subscripts defined within the Vensim model. Key is a string denoting the subscript name, values
 90        is a list of a sequence of arbitrary values for the subscript.
 91    state_vector_index_map : list[Union[None, StateVectorIndexMap]]
 92        This list holds the indices each stock variable values should go in if all stock variables were to be stored
 93        in a single vector. This is necessary because Stan requires that the input/output format of the ODE function
 94        be a single dimension vector. This means we need to take potentially multiple variables that are
 95        multidimensional, flatten them into one-dimensional vectors, and concatenate them. **Important**: Since stan
 96        indexing starts from 1, the 0'th index has `None` to indicate that it is not used.
 97    """
 98    def __init__(self, vensim_model_path: Union[str, pathlib.Path]) -> None:
 99        """
100        Parameters
101        ----------
102        vensim_model_path : string or pathlib.Path
103        """
104        vf = VensimFile(vensim_model_path)
105        vf.parse()
106        self._abstract_model: AbstractModel = vf.get_abstract_model()
107        self.variables: dict[str, VensimVariableContext] = {}
108        self.integ_outcome_variables: dict[str, VensimVariableContext] = {}
109        self._first_section: AbstractSection = self._abstract_model.sections[0]
110        self.state_vector_index_map: list[StateVectorIndexMap] = [None]
111
112        # Some basic checks to make sure the AM is compatible
113        assert len(self._abstract_model.sections) == 1, "Number of sections in AbstractModel must be 1."
114
115        for element in self._abstract_model.sections[0].elements:
116            element.name = vensim_name_to_identifier(element.name)
117            # Save names of variables defined in the model
118            assert len(element.components) == 1, f"Number of components in AbstractElement must be 1, but {element.name} has {len(element.components)}"
119
120            component = element.components[0]
121            subscripts = tuple(component.subscripts[0])  # Get the subscripts defined for the variable.
122            # See https://pysd.readthedocs.io/en/master/structure/vensim_translation.html#pysd.translators.vensim.vensim_element.Component
123
124            # Check if the variable is just static data
125            is_static_data = isinstance(component.ast, (int, float, np.ndarray))
126
127            # Check if the variable is a lookup
128            is_lookup = isinstance(component.ast, LookupsStructure)
129
130            # Check if the variable is a Vensim datastructure
131            is_datastructure = isinstance(component.ast, DataStructure)
132
133            variable_name = vensim_name_to_identifier(element.name)
134            is_stock = isinstance(component.ast, IntegStructure)
135            self.variables[variable_name] = VensimVariableContext(vensim_name_to_identifier(element.name), subscripts,
136                                                                  is_stock=is_stock, is_static_data=is_static_data,
137                                                                  is_lookup=is_lookup,
138                                                                  is_vensim_datastructure=is_datastructure)
139
140            # Additionally record the stock variables
141            if is_stock:
142                self.integ_outcome_variables[variable_name] = VensimVariableContext(vensim_name_to_identifier(element.name),
143                                                                                    subscripts, is_stock=True)
144
145        # record values of subscripts
146        self.subscripts: dict[str, tuple[Any]] = {}  # holds the values of subscripts defined within the model.
147        for subscript in self._first_section.subscripts:
148            self.subscripts[subscript.name] = tuple(subscript.subscripts)
149
150        # We create the mapping that maps subscripted, multidimensional stock variables into a single length vector.
151        for var_name, var_context in self.integ_outcome_variables.items():
152            subscript_dim = self.get_variable_shape(var_name)
153
154            # Calculate all possibles indices. If a subscript dimension is (2, 2), this means (1, 1), (1, 2),
155            # (2, 1), (2, 2) is possible.
156            subscript_indices = product(*[range(1, x + 1) for x in subscript_dim])
157
158            for indices in subscript_indices:
159                self.state_vector_index_map.append(StateVectorIndexMap(var_name, indices))
160
161    def set_timesteps_subscript(self, value: tuple[Number]) -> None:
162        """
163        This function is used to register the "timesteps" subscript. The timesteps subscript is used for stock variables
164        to indicate the time index.
165
166        Parameters
167        ----------
168        value : tuple[Number]
169            values of the subscript
170        """
171        self.subscripts["timesteps"] = value

174    def first_section(self) -> AbstractSection:
175        return self._first_section

178    def abstract_model(self) -> AbstractModel:
179        return self._abstract_model

181    def get_subscript_values(self, subscript_name: str) -> tuple[Any]:
182        """
183        Return the values of a given subscript.
184
185        Parameters
186        ----------
187        subscript_name : str
188            Subscript name
189
190        Returns
191        -------
192        A tuple with the values of the subscripts
193        """
194        return tuple(self.subscripts[subscript_name])

196    def get_variable_shape(self, variable_name: str) -> tuple[int]:
197        """
198        Return the required dimension of vensim variables.
199
200        If a Vensim variable is equipped with a subscript 'A' that has 2 values, it would need to be allocated as a
201        length-2 vector. In this case, this method will return `(2, )` indicating what its shape should be.
202
203        Parameters
204        ----------
205        variable_name : str
206            Variable name, must be an identifier
207
208        Returns
209        -------
210        A tuple of integer(s) indicating its required shape.
211        """
212        variable_context = self.variables[variable_name]
213        shape = []
214        for subscript in variable_context.subscripts:
215            shape.append(len(self.subscripts[subscript]))
216
217        return tuple(shape)
218
219    def print_variable_info(self) -> None:
220        """
221        Print some debug information about the variables present in the Vensim model.
222        """
223        var_names = []
224        max_length = len("original name") + 1
225        for element in self._abstract_model.sections[0].elements:
226            is_stock = False
227            for component in element.components:
228                if isinstance(component.ast, IntegStructure):
229                    is_stock = True
230                    break
231
232            var_names.append((element.name, vensim_name_to_identifier(element.name), is_stock,))
233            max_length = max(max_length, len(element.name) + 1)
234
235        header = ("original name".ljust(max_length) + "stan variable name".ljust(max_length) + "is stock")
236        print(header)
237        print("-" * len(header))
238        for x in var_names:
239            print(x[0].ljust(max_length) + x[1].ljust(max_length) + ("V" if x[2] else ""))
