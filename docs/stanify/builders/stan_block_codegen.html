 24@dataclass
 25class StanBlockCodegen(ABC):
 26    """
 27    Abstract class for classes that generate code for Stan blocks.
 28
 29    Attributes
 30    ----------
 31    _code : IndentedString
 32        The `stanify.builders.utilities.IndentedString` object that holds the generated code for blocks. Includes the block declarations and braces
 33    block_name : str
 34        Name of the Stan block
 35    _omit_block : bool
 36        Whether the block designation and braces should be omitted. Defaults to False. Should only be used during
 37        draws2data and nothing else.
 38
 39    Methods
 40    -------
 41    generate(v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext)
 42        To be implemented by subclasses. Run the generation logic and write results into `_code`.
 43    code()
 44        Returns the code string generated.
 45    """
 46    block_name: str
 47    _omit_block: bool = field(default=False)
 48    _code: IndentedString = field(init=False, default_factory=IndentedString)
 49
 50    @property
 51    def omit_block(self) -> bool:
 52        return self._omit_block
 53
 54    @omit_block.setter
 55    def omit_block(self, value: bool) -> None:
 56        self._omit_block = value
 57
 58    def __post_init__(self):
 59        if not self._omit_block:
 60            # Set up the block - write the block declaration and opening braces.
 61            self._code += f"{self.block_name} {{\n"
 62            # Increment indent level by 1 since we're inside the block body.
 63            self._code.indent_level += 1
 64
 65    @abstractmethod
 66    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
 67                 stan_model_context: StanModelContext) -> None:
 68        raise NotImplementedError
 69
 70    @property
 71    def code(self):
 72        if not self._omit_block:
 73            return str(self._code) + "\n}\n"
 74        else:
 75            return str(self._code)

 79class TransformedDataCodegenVensimWalker(StanBlockCodegen, BaseVensimWalker):
 80    """
 81    Generates code for the `transformed data` Stan block. This block includes variable declarations
 82    for constant variables.
 83
 84    Used for:
 85    - data2draws
 86    - draws2data
 87    """
 88
 89    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
 90                 stan_model_context: StanModelContext, omit_sbc_data_variable: bool = False) -> None:
 91        """
 92
 93        Parameters
 94        ----------
 95        v2s_code_handler : Vensim2StanCodeHandler
 96            see `StanBlockCodegen`
 97        vensim_model_context : VensimModelContext
 98            see `StanBlockCodegen`
 99        stan_model_context : StanModelContext
100            see `StanBlockCodegen`
101        omit_sbc_data_variable : bool
102            If a SBC data variable(`stanify.builders.vensim2stan.V2SModelSettings`) is a Vensim static
103            data, will not generate code for it if the parameter is set to `True`. This should only be set to true
104            during data2draws, since during draws2data the `data_variable` option is irrelevant.
105        """
106        # Write initial time
107        self._code += f"real initial_time = {v2s_code_handler.v2s_settings.initial_time};\n"
108
109        self._code += "// number of integration timesteps\n"
110        self._code += f"int timesteps = {len(v2s_code_handler.v2s_settings.integration_times)};\n"
111
112        # Create the array that holds the ODE integration timesteps
113        self._code += f"array[timesteps] real integration_times = {{ {', '.join([str(time) for time in v2s_code_handler.v2s_settings.integration_times])} }};\n\n"
114
115        self._code += f"int n_odes = {len(vensim_model_context.state_vector_index_map) - 1};\n"
116
117        self._code += "///////////////\n"
118        self._code += "// subscripts\n"
119
120        # Create the variables that hold the range of each subscript
121        for subscript_name, values in vensim_model_context.subscripts.items():
122            # Don't generate for the special subscript "timesteps"
123            if subscript_name == "timesteps":
124                continue
125            subscript_comment = [f"{val} : {index}" for index, val in enumerate(values, 1)]
126            self._code += f"int {subscript_name} = {len(values)};  // ({', '.join(subscript_comment)});\n"
127
128        self._code += "///////////////\n\n"
129
130        self._code += "// static Vensim variables\n"
131
132        # Walk through the Vensim model
133        for element in vensim_model_context.first_section.elements:
134            if element.name == "initial_time":
135                warnings.warn("initial time was defined in vensim. It will be ignored in favor for the initial time setting passed to vensim2stan.")
136                continue
137
138            # Skip the SBC data variable if it's requested
139            if omit_sbc_data_variable and element.name == v2s_code_handler.v2s_settings.data_variable:
140                continue
141
142            # if some variables are written as parameters, don't generate code
143            if element.name in stan_model_context.parameter_variables:
144                continue
145
146            for component in element.components:
147                declared_subscripts = component.subscripts[0]
148                self.walk(component.ast, element.name, declared_subscripts)
149
150        # Calculate the initial values for the stock variables.
151        self._code += "\n"
152        self._code += "// initial stock values\n"
153
154        # Create the initial state vector. This will be passed to the ODE solver
155        self._code += f"vector[n_odes] initial_state;\n"
156
157        # Calculate the initial values for each stock
158        for stock_name, var_context in vensim_model_context.integ_outcome_variables.items():
159            stan_varname = f"{stock_name}_initial"
160            if var_context.subscripts:
161                stan_type = f"array[{', '.join(var_context.subscripts)}] real"
162            else:
163                stan_type = "real"
164
165            walker = InitialValueCodegenVensimWalker(v2s_code_handler, vensim_model_context, stan_model_context)
166            for element in vensim_model_context.first_section.elements:
167                if element.name == stock_name:
168                    for component in element.components:
169                        code = walker.walk(component.ast, element.name, tuple(component.subscripts[0]))
170                        self._code += f"{stan_type} {stan_varname} = {code};\n"
171                        break
172
173        # Map the initial values into the initial state vector
174        for index, index_obj in enumerate(vensim_model_context.state_vector_index_map):
175            # skip index 1
176            if index == 0:
177                continue
178
179            if index_obj.indices:
180                indices = f"[{', '.join(str(i) for i in index_obj.indices)}]"
181            else:
182                indices = ""
183            self._code += f"initial_state[{index}] = {index_obj.name}_initial{indices};\n"
184
185    def walk(self, component_ast: pysd_ast.AbstractSyntax, node_name: str, subscripts: list[str] = None,
186             current_precedence: int = 100) -> None:
187        # Find constant data declarations.
188        match component_ast:
189            case int():
190                if subscripts:
191                    self._code += f"array[{'. '.join(subscripts)}] int {node_name} = rep_array({component_ast}, {', '.join(subscripts)});\n"
192                else:
193                    self._code += f"int {node_name} = {component_ast};\n"
194            case float():
195                if subscripts:
196                    self._code += f"array[{'. '.join(subscripts)}] real {node_name} = rep_array({component_ast}, {', '.join(subscripts)});\n"
197                else:
198                    self._code += f"real {node_name} = {component_ast};\n"
199            case np.ndarray():
200                # Check if the shape of the array matches the declared subscripts
201                assert component_ast.shape == self.vensim_model_context.get_variable_shape(node_name), \
202                    f"Vensim node {node_name} has subscript(s) {subscripts} of dimension {self.vensim_model_context.get_variable_shape(node_name)}, its length not matching the shape of the declared value array with shape {component_ast.shape}"
203
204                stan_dtype = "int" if np.issubdtype(component_ast.dtype, np.integer) else "real"
205
206                # Stan requires that array values be initialized with braces, instead of brackets.
207                # We take the numpy array values, convert them to a python list, and then replace all brackets in the
208                # stringified list with braces.
209                stan_array_init = str(component_ast.tolist()).replace("[", "{").replace("]", "}")
210
211                self._code += f"array[{', '.join(subscripts)}] {stan_dtype} {node_name} = {stan_array_init};\n"

216    """
217    Generate code for the Parameters Stan block.
218
219    Used for:
220    - data2draws
221    """
222
223    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
224                 stan_model_context: StanModelContext) -> None:
225        # iterate through the variables belonging in the parameter block
226        for variable_name in stan_model_context.parameter_variables:
227            # If variable is a SBC data variable, don't make it a variable but instead it comes from data
228            if variable_name == v2s_code_handler.v2s_settings.data_variable:

230            variable_context = v2s_code_handler.declared_variables[variable_name]
231
232            # Determine dimension and constraints

240            if constraint_code:
241                constraint_code = "<" + constraint_code + ">"
242
243            # Default type is float. This may not be optimal, but it should be fine for now.
244            variable_type = f"array[{', '.join(subscripts)}] real" if subscripts else "real"
245
246            # Save the subscript order in stan model context
247            if subscripts:
248                stan_model_context.array_dims_subscript_map[variable_name] = tuple(subscripts)
249
250            self._code += f"{variable_type}{constraint_code} {variable_name};\n"

285@dataclass
286class ModelBlockStatementV2SWalker(Vensim2StanWalker):
287    """
288    This walker traverses through the V2S AST, and generates individual Stan sampling statements.
289
290    Attributes
291    ----------
292    subscript_loop_variable_mapping : dict[str, str]
293        Dict which maps subscript names to loop variable. `dict[subscript_name, loop_variable]` where `subscript_name`
294        is the original subscript name, and `loop_variable` is the Stan loop variable(`i, j, k, ...`) that corresponds
295        to it.
296    array_dims_subscript_map : dict[str, tuple[str]]
297        dict that holds the dimension order for each stan variable. Should be from
298        `stanify.builders.stan_model_context.StanModelContext`
299    _code : IndentedString
300        `stanify.builders.utilities.IndentedString` that holds the generated code
301
302    """
303    subscript_loop_variable_mapping: dict[str, str]
304    array_dims_subscript_map: dict[str, tuple[str]]
305    _code: IndentedString = field(init=False, default_factory=IndentedString)
306
307    def walk_Statement(self, node: ast.Statement):
308        if node.op == "=":
309            return
310        self.walk(node.left)
311        self._code += " ~ "
312        self.walk(node.right)
313        self._code += ";\n"
314
315    def walk_Binary(self, node: ast.Binary):
316        self.walk(node.left)
317        self._code += f" {node.op} "
318        self.walk(node.right)
319
320    def walk_FunctionCall(self, node: ast.FunctionCall):
321        self._code += f"{node.name}("
322        for index, arg in enumerate(node.arglist):
323            self.walk(arg)
324            if index < len(node.arglist) - 1:
325                self._code += ", "
326        self._code += ")"
327
328    def walk_Variable(self, node: ast.Variable):
329        self._code += node.name
330        if node.subscripts:
331            # Check that the info in the dims-subscript map is the same as the AST subscripts
332            assert set(node.subscripts) == set(self.array_dims_subscript_map[node.name]), f"Found unmatched subscript-dimension map information for Stan variable {node.name}"
333
334            # Sort the subscript order according to their declared ordering
335            subscript_order = sorted(list(node.subscripts), key=lambda x: self.array_dims_subscript_map[node.name].index(x))
336
337            self._code += "["
338            self._code += ",".join(self.subscript_loop_variable_mapping[subscript] for subscript in subscript_order)
339            self._code += "]"
340
341    def walk_Literal(self, node: ast.Literal):
342        self._code += f"{node.value}"
343
344    @property
345    def code(self) -> str:
346        return str(self._code)

349class ModelBlockCodegen(StanBlockCodegen):
350    """
351    Generates code for the `model` Stan block. This includes prior and likelihood specifications.
352
353    Used for:
354    - data2draws
355    """
356
357    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
358                 stan_model_context: StanModelContext) -> None:
359        # Find subscripted variables, and figure out the way to loop through them with the minimum amount of loops.
360        # If we have three variables, say with subscript usage (a, b), (b, c), (c), (a), it would be best to run like
361        # the following loop:
362        # for a in A:
363        #   (a)
364        #   for b in B:
365        #       (a, b)
366        #
367        # for c in C:
368        #   (c)
369        #   for b in B:
370        #       (b, c)
371        #
372        # cost = 2 * 2 + 2 * 2
373
374        # Problem definition - minimize the total cost of loops while ensuring all subscript combinations are used.
375        # This felt really similar to Quine-McClusky; I had to find the "combinations" that covered all subscripts while
376        # minimizing duplicate subscript loops.
377
378        # This set holds all the subscript combinations that need to be covered
379        used_subscripts = set()
380
381        # Iterate through all declared LHS-variables and record what subscript was used with what subscript.
382        for _, declared_variable in v2s_code_handler.declared_variables.items():
383            if declared_variable.subscripts:
384                used_subscripts.add(frozenset(declared_variable.subscripts))
385
386        # We will use a simple tree structure to hold the hierarchical order of nested for loops.
387        @dataclass
388        class Node:
389            value: str
390            children: set["Node"] = field(default_factory=set)
391
392            def __hash__(self):
393                return hash(self.value)
394
395        # We're going to recursively iterate through the subscript sets. This helps when we have triple-subscripted
396        # variables, and maybe even deeper!
397        # I'll try my best to write out my thought process
398
399        # This is the driving function
400        def find_order(subscript_combinations: list[frozenset[str]], parent: Node):
401            """
402
403            Parameters
404            ----------
405            subscript_combinations : list[set[str]]
406                The current subscripts that we need to find out the ordering for
407            parent : Node
408                parent Node
409            """
410
411            # For each single subscript value, record what subscript combinations contain the value
412            usages: dict[str, set[frozenset[str]]] = defaultdict(set)
413
414            for combination in subscript_combinations:
415                for subscript in combination:
416                    usages[subscript].add(frozenset(combination))
417
418            # We sort the subscript values base on the most used subscript values.
419            sorted_subscripts = sorted(usages.items(), key=lambda x: len(x[1]), reverse=True)
420
421            # This set holds the subscript combinations that are already covered
422            covered = set()
423
424            # Iterate through The usage dictionary, checking whether we need to add the subscript value
425            for subscript, combinations in sorted_subscripts:
426                required = False
427                current_node = Node(subscript)
428
429                # This list holds the subscript combination with the current subscript value removed.
430                # For example, if we had (a, b, c) and (a, c) and we're currently at a, it would contain
431                # (b, c) and (c,)
432                self_removed = []
433
434                for combination in combinations:
435                    # If there is a subscript combination that's not yet covered, we need the current subscript value
436                    if combination not in covered:
437                        required = True
438                    else:
439                        continue
440                    covered.add(combination)
441
442                    # Create the subscript combination with current value removed
443                    subtracted = frozenset(combination - {subscript})
444                    if subtracted:
445                        self_removed.append(subtracted)
446
447                # We run the functoin again, but this time with the child subscripts(that doesnn't have the current
448                # value anymore), and parent becomes the current subscript value.
449                find_order(self_removed, current_node)
450
451                # We only add it to the parent if it's required(the subscript has a new combination)
452                if required:
453                    parent.children.add(current_node)
454
455        # Create a dummy node to hold the results
456        root = Node("")
457        find_order(list(used_subscripts), root)
458
459        # Now we run code generation by traversing through the V2S AST
460        # Keep a set of V2S variables we've already processed
461        completed_v2s_variables = set()
462
463        def build_forloops(node: Node, current_subscripts: dict[str, str], nest_level=0):
464            """
465            Parameters
466            ----------
467            node : Node
468            current_subscripts : dict[str, str]
469                dict indicating the current subscripts and their loop bound variables
470                `dict[subscript_name, loop_bound]` where `subscript_name` is the subscript name and `loop_bound` is the
471                loop bounded variable (i, j, k, ...)
472            nest_level : current nest level of the for loops. base starting value is 0(outermost for loop)
473            """
474            if not node:
475                return
476            loop_bound_variable = chr(ord("i") + nest_level)  # i, j, k, l, ...
477            current_subscripts[node.value] = loop_bound_variable
478
479            current_vensim_subscripts = set(current_subscripts.keys())
480
481            self._code += f"for ({loop_bound_variable} in 1:{node.value}) {{\n"
482            self._code.indent_level += 1
483
484            for statement in v2s_code_handler.program_ast.statements:
485                if statement.op != "~":
486                    continue
487
488                lhs_variable = statement.left
489
490                if lhs_variable.subscripts:
491                    if set(lhs_variable.subscripts) == current_vensim_subscripts and lhs_variable.name not in completed_v2s_variables:
492                        completed_v2s_variables.add(lhs_variable.name)
493                        stmt_walker = ModelBlockStatementV2SWalker(current_subscripts,
494                                                                   stan_model_context.array_dims_subscript_map)
495                        stmt_walker.walk(statement)
496                        self._code += stmt_walker.code
497
498            for children in node.children:
499                build_forloops(children, current_subscripts, nest_level + 1)
500            self._code.indent_level -= 1
501            self._code += "}\n"
502
503        for child in root.children:
504            build_forloops(child, {})
505
506        # After we handled the subscripted statements, handle the non-subscripted statements.
507        for statement in v2s_code_handler.program_ast.statements:
508            if statement.op != "~":
509                continue
510
511            lhs_variable = statement.left
512            if not lhs_variable.subscripts and lhs_variable.name not in completed_v2s_variables:
513                completed_v2s_variables.add(lhs_variable.name)
514                walker = ModelBlockStatementV2SWalker({}, stan_model_context.array_dims_subscript_map)
515                walker.walk(statement)
516                self._code += walker.code
