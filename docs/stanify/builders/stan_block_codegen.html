 25@dataclass
 26class StanBlockCodegen(ABC):
 27    """
 28    Abstract class for classes that generate code for Stan blocks.
 29
 30    Attributes
 31    ----------
 32    _code : IndentedString
 33        The `stanify.builders.utilities.IndentedString` object that holds the generated code for blocks. Includes the block declarations and braces
 34    block_name : str
 35        Name of the Stan block
 36    _omit_block : bool
 37        Whether the block designation and braces should be omitted. Defaults to False. Should only be used during
 38        draws2data and nothing else.
 39
 40    Methods
 41    -------
 42    generate(v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext)
 43        To be implemented by subclasses. Run the generation logic and write results into `_code`.
 44    code()
 45        Returns the code string generated.
 46    """
 47    block_name: str
 48    _omit_block: bool = field(default=False)
 49    _code: IndentedString = field(init=False, default_factory=IndentedString)
 50
 51    @property
 52    def omit_block(self) -> bool:
 53        return self._omit_block
 54
 55    @omit_block.setter
 56    def omit_block(self, value: bool) -> None:
 57        self._omit_block = value
 58
 59    def __post_init__(self):
 60        if not self._omit_block:
 61            # Set up the block - write the block declaration and opening braces.
 62            self._code += f"{self.block_name} {{\n"
 63            # Increment indent level by 1 since we're inside the block body.
 64            self._code.indent_level += 1
 65
 66    @abstractmethod
 67    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
 68                 stan_model_context: StanModelContext) -> None:
 69        raise NotImplementedError
 70
 71    @property
 72    def code(self):
 73        # Add the block enclosing braces
 74        if not self._omit_block:
 75            return str(self._code) + "\n}\n"
 76        else:
 77            return str(self._code)

 81class InitialValueStatementsCodegen:
 82    """
 83    Generates code for statements needed for calculating initial values
 84    """
 85    _code: IndentedString = field(init=False, default_factory=IndentedString)
 86    requires_params: bool = field(init=False, default=True)
 87    required_params: set[str] = field(init=False, default_factory=set)
 88
 89    @property
 90    def code(self) -> str:
 91        return str(self._code)
 92
 93    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
 94                 stan_model_context: StanModelContext) -> None:
 95        # Calculate the initial values for the stock variables.
 96        self._code += "\n"
 97        self._code += "// initial stock values\n"
 98
 99        # Create the initial state vector. This will be passed to the ODE solver
100        self._code += f"vector[n_odes] initial_state;\n"
101
102        # Find the required variables needed, and the topological order
103        ignored_variables = set()
104        ignored_variables.update(stan_model_context.lookupfunc_variables)
105        ignored_variables.update(stan_model_context.timestep_variant_datafunc_variables)
106        ignored_variables.update(stan_model_context.transformed_data_variables)
107
108        statement_sorter = StatementTopoSort(tuple(ignored_variables))
109
110        walker = FindODEInitialRHSVariablesVensimWalker()
111        for element in vensim_model_context.first_section.elements:
112            lhs_variable = element.name
113            for component in element.components:
114                rhs_variables = walker.walk(component.ast)
115
116                statement_sorter.add_stmt(lhs_variable, rhs_variables)
117
118        required_variables = set()
119        for stock_name in vensim_model_context.integ_outcome_variables.keys():
120            required_variables |= statement_sorter.find_required_variables(stock_name)
121
122        if not required_variables.isdisjoint(stan_model_context.parameter_variables):
123            self.requires_params = True
124            self.required_params = required_variables.intersection(stan_model_context.parameter_variables)
125
126        statement_eval_order = statement_sorter.sort()
127        elements = [element for element in vensim_model_context.first_section.elements if
128                    element.name in statement_eval_order]
129
130        elements = sorted(elements, key=lambda x: statement_eval_order.index(x.name))
131        for element in elements:
132            name = element.name
133
134            # Only codegen for variables required for the initial values
135            if name not in required_variables:
136                continue
137
138            # Variables declared in stan can be skipped
139            if name in stan_model_context.get_stan_declared_variables():
140                continue
141
142            var_context = vensim_model_context.variables[name]
143            walker = InitialValueCodegenVensimWalker(v2s_code_handler, vensim_model_context, stan_model_context)
144
145            if var_context.is_stock:
146                stan_varname = f"{name}_initial"
147            else:
148                stan_varname = name
149
150            if var_context.subscripts:
151                stan_type = f"array[{', '.join(var_context.subscripts)}] real"
152            else:
153                stan_type = "real"
154
155            for component in element.components:
156                used_subscripts, excluded_subscripts = component.subscripts
157                code = walker.walk(component.ast, element.name, tuple(used_subscripts))
158                self._code += f"{stan_type} {stan_varname} = {code};\n"
159                break
160
161        # Map the initial values into the initial state vector
162        for index, index_obj in enumerate(vensim_model_context.state_vector_index_map):
163            # skip index 1
164            if index == 0:
165                continue
166
167            if index_obj.indices:
168                indices = f"[{', '.join(str(i) for i in index_obj.indices)}]"
169            else:
170                indices = ""
171            self._code += f"initial_state[{index}] = {index_obj.name}_initial{indices};\n"

175class TransformedDataCodegenVensimWalker(StanBlockCodegen):
176    """
177    Generates code for the `transformed data` Stan block. This block includes variable declarations
178    for constant variables and the initial values for the ODE system.
179
180    Used for:
181    - data2draws
182    - draws2data
183    """
184
185    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
186                 stan_model_context: StanModelContext, omit_sbc_data_variable: bool = False) -> None:
187        """
188
189        Parameters
190        ----------
191        v2s_code_handler : Vensim2StanCodeHandler
192            see `StanBlockCodegen`
193        vensim_model_context : VensimModelContext
194            see `StanBlockCodegen`
195        stan_model_context : StanModelContext
196            see `StanBlockCodegen`
197        omit_sbc_data_variable : bool
198            If a SBC data variable(`stanify.builders.vensim2stan.V2SModelSettings`) is a Vensim static
199            data, will not generate code for it if the parameter is set to `True`. This should only be set to true
200            during data2draws, since during draws2data the `data_variable` option is irrelevant.
201        """
202        # Write initial time
203        self._code += f"real initial_time = {v2s_code_handler.v2s_settings.initial_time};\n"
204
205        self._code += "// number of integration timesteps\n"
206        self._code += f"int timesteps = {len(v2s_code_handler.v2s_settings.integration_times)};\n"
207
208        # Create the array that holds the ODE integration timesteps
209        self._code += f"array[timesteps] real integration_times = {{ {', '.join([str(time) for time in v2s_code_handler.v2s_settings.integration_times])} }};\n\n"
210
211        self._code += f"int n_odes = {len(vensim_model_context.state_vector_index_map) - 1};\n"
212
213        self._code += "///////////////\n"
214        self._code += "// subscripts\n"
215
216        # Create the variables that hold the range of each subscript
217        for subscript_name, values in vensim_model_context.subscripts.items():
218            # Don't generate for the special subscript "timesteps"
219            if subscript_name == "timesteps":
220                continue
221            subscript_comment = [f"{val} : {index}" for index, val in enumerate(values, 1)]
222            self._code += f"int {subscript_name} = {len(values)};  // ({', '.join(subscript_comment)});\n"
223
224        self._code += "///////////////\n\n"
225
226        self._code += "// static Vensim variables\n"
227
228        # Write static data variables that are timestep-invariant
229        for variable_name in stan_model_context.transformed_data_variables:
230            if variable_name not in stan_model_context.static_data_variable_values:
231                raise Exception(f"Variable value for '{variable_name}' is not present!!")
232
233            variable_value = stan_model_context.static_data_variable_values[variable_name]
234
235            if variable_name == "initial_time":
236                warnings.warn("initial time was defined in vensim. It will be ignored in favor for the initial time setting passed to Vensim2Stan.")
237                continue
238
239            # Skip the SBC data variable if it's requested
240            if omit_sbc_data_variable and variable_name == v2s_code_handler.v2s_settings.data_variable:
241                continue
242
243            match variable_value:
244                case int():
245                    self._code += f"int {variable_name} = {variable_value};\n"
246                case float():
247                    self._code += f"real {variable_name} = {variable_value};\n"
248                case xarray.DataArray():
249                    stan_dtype = "int" if np.issubdtype(variable_value.dtype, np.integer) else "real"
250                    subscripts = (str(subscript) for subscript in variable_value.dims)
251
252                    # Stan requires that array values be initialized with braces, instead of brackets.
253                    # We take the numpy array values, convert them to a python list, and then replace all brackets in the
254                    # string-ified list with braces.
255                    stan_array_init = str(variable_value.data.tolist()).replace("[", "{").replace("]", "}")
256
257                    self._code += f"array[{', '.join(subscripts)}] {stan_dtype} {variable_name} = {stan_array_init};\n"

264

269    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
270                 stan_model_context: StanModelContext) -> None:
271        # iterate through the variables belonging in the parameter block
272        for variable_name in stan_model_context.parameter_variables:
273            # If variable is a SBC data variable, don't make it a variable but instead it comes from data
274            if variable_name == v2s_code_handler.v2s_settings.data_variable:

277
278            # Determine dimension and constraints

282                constraint_code += f"lower={variable_context.lower}"

286            if constraint_code:
287                constraint_code = "<" + constraint_code + ">"
288
289            # Default type is float. This may not be optimal, but it should be fine for now.
290            variable_type = f"array[{', '.join(subscripts)}] real" if subscripts else "real"
291
292            # Save the subscript order in stan model context
293            if subscripts:
294                stan_model_context.array_dims_subscript_map[variable_name] = tuple(subscripts)
295
296            self._code += f"{variable_type}{constraint_code} {variable_name};\n"

332@dataclass
333class ModelBlockStatementV2SWalker(Vensim2StanWalker):
334    """
335    This walker traverses through the V2S AST, and generates individual Stan sampling statements.
336
337    Attributes
338    ----------
339    subscript_loop_variable_mapping : dict[str, str]
340        Dict which maps subscript names to loop variable. `dict[subscript_name, loop_variable]` where `subscript_name`
341        is the original subscript name, and `loop_variable` is the Stan loop variable(`i, j, k, ...`) that corresponds
342        to it.
343    array_dims_subscript_map : dict[str, tuple[str]]
344        dict that holds the dimension order for each stan variable. Should be from
345        `stanify.builders.stan_model_context.StanModelContext`
346    _code : IndentedString
347        `stanify.builders.utilities.IndentedString` that holds the generated code
348
349    """
350    subscript_loop_variable_mapping: dict[str, str]
351    array_dims_subscript_map: dict[str, tuple[str]]
352    _code: IndentedString = field(init=False, default_factory=IndentedString)
353
354    def walk_Statement(self, node: ast.Statement):
355        if node.op == "=":
356            return
357        self.walk(node.left)
358        self._code += " ~ "
359        self.walk(node.right)
360        self._code += ";\n"
361
362    def walk_Binary(self, node: ast.Binary):
363        self.walk(node.left)
364        self._code += f" {node.op} "
365        self.walk(node.right)
366
367    def walk_FunctionCall(self, node: ast.FunctionCall):
368        self._code += f"{node.name}("
369        for index, arg in enumerate(node.arglist):
370            self.walk(arg)
371            if index < len(node.arglist) - 1:
372                self._code += ", "
373        self._code += ")"
374
375    def walk_Variable(self, node: ast.Variable):
376        self._code += node.name
377        if node.subscripts:
378            # Check that the info in the dims-subscript map is the same as the AST subscripts
379            assert set(node.subscripts) == set(self.array_dims_subscript_map[node.name]), f"Found unmatched subscript-dimension map information for Stan variable {node.name}"
380
381            # Sort the subscript order according to their declared ordering
382            subscript_order = sorted(list(node.subscripts), key=lambda x: self.array_dims_subscript_map[node.name].index(x))
383
384            self._code += "["
385            self._code += ",".join(self.subscript_loop_variable_mapping[subscript] for subscript in subscript_order)
386            self._code += "]"
387
388    def walk_Literal(self, node: ast.Literal):
389        self._code += f"{node.value}"
390
391    @property
392    def code(self) -> str:
393        return str(self._code)

396class ModelBlockCodegen(StanBlockCodegen):
397    """
398    Generates code for the `model` Stan block. This includes prior and likelihood specifications.
399
400    Used for:
401    - data2draws
402    """
403
404    def generate(self, v2s_code_handler: Vensim2StanCodeHandler, vensim_model_context: VensimModelContext,
405                 stan_model_context: StanModelContext) -> None:
406        # Find subscripted variables, and figure out the way to loop through them with the minimum amount of loops.
407        # If we have three variables, say with subscript usage (a, b), (b, c), (c), (a), it would be best to run like
408        # the following loop:
409        # for a in A:
410        #   (a)
411        #   for b in B:
412        #       (a, b)
413        #
414        # for c in C:
415        #   (c)
416        #   for b in B:
417        #       (b, c)
418        #
419        # cost = 2 * 2 + 2 * 2
420
421        # Problem definition - minimize the total cost of loops while ensuring all subscript combinations are used.
422        # This felt really similar to Quine-McClusky; I had to find the "combinations" that covered all subscripts while
423        # minimizing duplicate subscript loops.
424
425        # This set holds all the subscript combinations that need to be covered
426        used_subscripts = set()
427
428        # Iterate through all declared LHS-variables and record what subscript was used with what subscript.
429        for _, declared_variable in v2s_code_handler.declared_variables.items():
430            if declared_variable.subscripts:
431                used_subscripts.add(frozenset(declared_variable.subscripts))
432
433        # We will use a simple tree structure to hold the hierarchical order of nested for loops.
434        @dataclass
435        class Node:
436            value: str
437            children: set["Node"] = field(default_factory=set)
438
439            def __hash__(self):
440                return hash(self.value)
441
442        # We're going to recursively iterate through the subscript sets. This helps when we have triple-subscripted
443        # variables, and maybe even deeper!
444        # I'll try my best to write out my thought process
445
446        # This is the driving function
447        def find_order(subscript_combinations: list[frozenset[str]], parent: Node):
448            """
449
450            Parameters
451            ----------
452            subscript_combinations : list[set[str]]
453                The current subscripts that we need to find out the ordering for
454            parent : Node
455                parent Node
456            """
457
458            # For each single subscript value, record what subscript combinations contain the value
459            usages: dict[str, set[frozenset[str]]] = defaultdict(set)
460
461            for combination in subscript_combinations:
462                for subscript in combination:
463                    usages[subscript].add(frozenset(combination))
464
465            # We sort the subscript values base on the most used subscript values.
466            sorted_subscripts = sorted(usages.items(), key=lambda x: len(x[1]), reverse=True)
467
468            # This set holds the subscript combinations that are already covered
469            covered = set()
470
471            # Iterate through The usage dictionary, checking whether we need to add the subscript value
472            for subscript, combinations in sorted_subscripts:
473                required = False
474                current_node = Node(subscript)
475
476                # This list holds the subscript combination with the current subscript value removed.
477                # For example, if we had (a, b, c) and (a, c) and we're currently at a, it would contain
478                # (b, c) and (c,)
479                self_removed = []
480
481                for combination in combinations:
482                    # If there is a subscript combination that's not yet covered, we need the current subscript value
483                    if combination not in covered:
484                        required = True
485                    else:
486                        continue
487                    covered.add(combination)
488
489                    # Create the subscript combination with current value removed
490                    subtracted = frozenset(combination - {subscript})
491                    if subtracted:
492                        self_removed.append(subtracted)
493
494                # We run the function again, but this time with the child subscripts(that doesn't have the current
495                # value anymore), and parent becomes the current subscript value.
496                find_order(self_removed, current_node)
497
498                # We only add it to the parent if it's required(the subscript has a new combination)
499                if required:
500                    parent.children.add(current_node)
501
502        # Create the root node to hold the results
503        root = Node("")
504        find_order(list(used_subscripts), root)
505
506        # Now we run code generation by traversing through the V2S AST
507        # Keep a set of V2S variables we've already processed
508        completed_v2s_variables = set()
509
510        def build_forloops(node: Node, current_subscripts: dict[str, str], nest_level=0):
511            """
512            Parameters
513            ----------
514            node : Node
515            current_subscripts : dict[str, str]
516                dict indicating the current subscripts and their loop bound variables
517                `dict[subscript_name, loop_bound]` where `subscript_name` is the subscript name and `loop_bound` is the
518                loop bounded variable (i, j, k, ...)
519            nest_level : current nest level of the for loops. base starting value is 0(outermost for loop)
520            """
521            if not node:
522                return
523            loop_bound_variable = chr(ord("i") + nest_level)  # i, j, k, l, ...
524            current_subscripts[node.value] = loop_bound_variable
525
526            current_vensim_subscripts = set(current_subscripts.keys())
527
528            self._code += f"for ({loop_bound_variable} in 1:{node.value}) {{\n"
529            self._code.indent_level += 1
530
531            for statement in v2s_code_handler.program_ast.statements:
532                if statement.op != "~":
533                    continue
534
535                lhs_variable = statement.left
536
537                if lhs_variable.subscripts:
538                    if set(lhs_variable.subscripts) == current_vensim_subscripts and lhs_variable.name not in completed_v2s_variables:
539                        completed_v2s_variables.add(lhs_variable.name)
540                        stmt_walker = ModelBlockStatementV2SWalker(current_subscripts,
541                                                                   stan_model_context.array_dims_subscript_map)
542                        stmt_walker.walk(statement)
543                        self._code += stmt_walker.code
544
545            for children in node.children:
546                build_forloops(children, current_subscripts, nest_level + 1)
547            self._code.indent_level -= 1
548            self._code += "}\n"
549
550        for child in root.children:
551            build_forloops(child, {})
552
553        # After we handled the subscripted statements, handle the non-subscripted statements.
554        for statement in v2s_code_handler.program_ast.statements:
555            if statement.op != "~":
556                continue
557
558            lhs_variable = statement.left
559            if not lhs_variable.subscripts and lhs_variable.name not in completed_v2s_variables:
560                completed_v2s_variables.add(lhs_variable.name)
561                walker = ModelBlockStatementV2SWalker({}, stan_model_context.array_dims_subscript_map)
562                walker.walk(statement)
563                self._code += walker.code

628            statement_sorter.add_stmt(stock_varname, stan_model_context.odefunc_variable_args)

630        for statement in v2s_code_handler.program_ast.statements:
631            if statement.op == "=":
632                continue
633
634            left_variable = statement.left
635
636            # If the LHS is a data variable, we ignore the sample statement
637            if left_variable.not_param:

643            rhs_variables = rhs_variable_walker.variable_names
644
645            statement_sorter.add_stmt(left_variable.name, rhs_variables)
646
647        param_draw_order = statement_sorter.sort()
648
649        # Generate the initial ODE values, but don't write them yet
650        initial_values_codegen = InitialValueStatementsCodegen()
651        initial_values_codegen.generate(v2s_code_handler, vensim_model_context, stan_model_context)
652
653        # these are the parameters that are drawn before running the ODE. We populate them first with parameters that
654        # are needed for the initial values.
655        pre_ode_run_params = list(initial_values_codegen.required_params)
656        param_draw_order = [param for param in param_draw_order if param not in pre_ode_run_params]
657
658        post_ode_run_params = []  # these are the parameters that are drawn after running the ODE
659        # Find all parameters before any stock variables
660        for param_name in param_draw_order:
661            param_draw_order.remove(param_name)
662            if param_name in vensim_model_context.integ_outcome_variables:
663                break
664
665            pre_ode_run_params.append(param_name)
666
667        # Once we see a stock variable, remove the remaining stock variables and the rest can be drawn after the ODE
668        for param_name in param_draw_order:
669            if param_name in vensim_model_context.integ_outcome_variables:
670                param_draw_order.remove(param_name)
671
672        post_ode_run_params = param_draw_order[:]
673
674        # Iterate through the pre ODE statements in the sorted order
675        for param_name in pre_ode_run_params:
676            if param_name not in stan_model_context.parameter_variables:
677                continue
678
679            for statement in v2s_code_handler.program_ast.statements:
680                left_variable = statement.left
681                left_variable_name = statement.left.name
682                if left_variable_name != param_name:
683                    continue
684
685                if left_variable.subscripts:
686                    subscript_args = ", ".join(left_variable.subscripts)
687                    stan_dtype = f"array[{subscript_args}] real"
688                    stan_model_context.array_dims_subscript_map[left_variable_name] = tuple(left_variable.subscripts)
689                else:
690                    stan_dtype = "real"
691
692                self._code += f"{stan_dtype} {param_name};\n"
693
694                self.generate_code_for_statements(statement, v2s_code_handler, vensim_model_context, stan_model_context)
695
696        # Write the initial value code
697        self._code += initial_values_codegen.code
698        self._code += "\n"
699
700        # generate the ODE statements. We do this through the transformed parameters code generator.
701        transformed_parameters_gen = TransformedParametersBlockCodegen("transformed parameters", _omit_block=True)
702        # omit the block and braces stuff
703        # manually set the indent level to 1
704        transformed_parameters_gen._code.indent_level = 1
705        transformed_parameters_gen.generate(v2s_code_handler, vensim_model_context, stan_model_context)
706        self._code.add_raw(transformed_parameters_gen.code, ignore_indent=True)
707        self._code += "\n"
708
709        # Now generate the post-ODE statements
710        for param_name in post_ode_run_params:
711            if param_name not in stan_model_context.parameter_variables:
712                continue
713
714            for statement in v2s_code_handler.program_ast.statements:
715                left_variable = statement.left
716                left_variable_name = statement.left.name
717                if left_variable_name != param_name:
718                    continue
719
720                if left_variable.subscripts:
721                    subscript_args = ", ".join(left_variable.subscripts)
722                    stan_dtype = f"array[{subscript_args}] real"
723                    stan_model_context.array_dims_subscript_map[left_variable_name] = tuple(left_variable.subscripts)
724                else:
725                    stan_dtype = "real"
726
727                self._code += f"{stan_dtype} {param_name};\n"
728
729                self.generate_code_for_statements(statement, v2s_code_handler, vensim_model_context, stan_model_context)
730
731    def generate_code_for_statements(self, statement: ast.Statement, v2s_code_handler: Vensim2StanCodeHandler,
732                                     vensim_model_context: VensimModelContext, stan_model_context: StanModelContext):
733        if statement.op == "=":
734            return
735
736        left_variable = statement.left
737
738        loop_variable_mapping = {}  # key is subscript name, value is loop variable
739        if left_variable.subscripts:
740            indent_levels = len(left_variable.subscripts)
741
742            for nest_level in range(indent_levels):
743                loop_variable = chr(ord("i") + nest_level)
744                loop_bound = left_variable.subscripts[nest_level]
745                self._code += f"for ({loop_variable} in 1:{loop_bound}){{\n"
746                self._code.indent_level += 1
747                loop_variable_mapping[loop_bound] = loop_variable

749        walker = Draws2DataGeneratedQuantitiesDrawV2SWalker(loop_variable_mapping,
750                                                            stan_model_context.array_dims_subscript_map)
751        walker.walk(statement)
752        self._code += walker.code
753
754        if left_variable.subscripts:
755            for nest_level in range(len(left_variable.subscripts)):
756                self._code.indent_level -= 1
757                self._code += "}\n"

407        # Find subscripted variables, and figure out the way to loop through them with the minimum amount of loops.
408        # If we have three variables, say with subscript usage (a, b), (b, c), (c), (a), it would be best to run like
409        # the following loop:
410        # for a in A:
411        #   (a)
412        #   for b in B:
413        #       (a, b)

423        # This felt really similar to Quine-McClusky; I had to find the "combinations" that covered all subscripts while
424        # minimizing duplicate subscript loops.
